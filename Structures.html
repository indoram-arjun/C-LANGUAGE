Structures in C

What is a Structure?

A structure in C is a user-defined data type that groups different types of data under one name. ðŸ‘‰ It is like a container where we can store variables of different data types together.

Printing a String

  • Example: A student has â†’ name (string), age (int), marks (float).
  • Using structure, we can keep all these in one place.

Syntax of Structure


            struct structure_name {
            dataType member1;
            dataType member2;
            ...
            };

Example 1: Simple Structure


            #include <stdio.h>
            // defining structure
            struct Student {
                char name[20];
                int age;
                float marks;
            };

            int main() {
                struct Student s1;   // declaring structure variable

                // assigning values
                strcpy(s1.name, "Arjun");
                s1.age = 22;
                s1.marks = 87.5;

                // printing values
                printf("Name: %s\n", s1.name);
                printf("Age: %d\n", s1.age);
                printf("Marks: %.2f", s1.marks);

                return 0;
            }

Output:


            Name: Arjun
            Age: 22
            Marks: 87.50

Example 2: Array of Structures

We can store multiple students in an array.


            #include <stdio.h>
            #include <string.h>

                struct Student {
                    char name[20];
                    int age;
                    float marks;
                };

                int main() {
                    struct Student s[2];  // array of structures

                    // student 1
                    strcpy(s[0].name, "Harsh");
                    s[0].age = 20;
                    s[0].marks = 90.5;

                    // student 2
                    strcpy(s[1].name, "Arjun");
                    s[1].age = 22;
                    s[1].marks = 85.0;

                    // printing
                    for(int i=0; i< 2; i++) {
                        printf("\nName: %s, Age: %d, Marks: %.2f", s[i].name, s[i].age, s[i].marks);
                    }

                    return 0;
                }

Output:


            Name: Harsh, Age: 20, Marks: 90.50
            Name: Arjun, Age: 22, Marks: 85.00

Example 3: Nested Structures

Structure inside another structure.


            #include <stdio.h>
            struct Address {
                char city[20];
                int pin;
            };

            struct Student {
                char name[20];
                int age;
                struct Address addr;   // nested structure
            };

            int main() {
                struct Student s1 = {"Arjun", 22, {"Mumbai", 400001}};

                printf("Name: %s\n", s1.name);
                printf("Age: %d\n", s1.age);
                printf("City: %s\n", s1.addr.city);
                printf("Pin: %d", s1.addr.pin);

                return 0;
            }

Output:


            Name: Arjun
            Age: 22
            City: Mumbai
            Pin: 400001

Key Points
